onst express = require("express");
const { BetaAnalyticsDataClient } = require("@google-analytics/data");
const app = express();
const port = 3001;

const analyticsDataClient = new BetaAnalyticsDataClient({
  keyFilename: "chemin/vers/ta-clé-service.json",
});

const propertyId = "GA4_PROPERTY_ID"; // Remplace avec ton ID Google Analytics 4

app.get("/api/analytics", async (req, res) => {
  try {
    const [response] = await analyticsDataClient.runReport({
      property: `properties/${propertyId}`,
      dateRanges: [{ startDate: "7daysAgo", endDate: "today" }],
      dimensions: [{ name: "date" }],
      metrics: [
        { name: "screenPageViews" },
        { name: "activeUsers" },
        { name: "bounceRate" },
      ],
    });

    const stats = response.rows.map((row) => ({
      date: row.dimensionValues[0].value,
      vues: row.metricValues[0].value,
      utilisateurs: row.metricValues[1].value,
      bounceRate: row.metricValues[2].value,
    }));

    res.json(stats);
  } catch (err) {
    console.error(err);
    res.status(500).send("Erreur récupération Analytics");
  }
});

app.listen(port, () => {
  console.log(`Serveur backend API GA tournant sur http://localhost:${port}`);
});
